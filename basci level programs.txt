int main() {

    int arr[n];

    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int max = arr[0];  // assume first element is max

    for (i = 1; i < n; i++) {
